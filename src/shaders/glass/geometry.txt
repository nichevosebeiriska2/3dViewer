#version 430 core

layout(triangles) in;
layout(triangle_strip, max_vertices = 3) out;

out vec3 fragColor;
out vec3 vecNormal;

uniform vec3 light_pos;
uniform mat4 mvp;
uniform mat4 model;
uniform vec3 my_position;
uniform vec3 model_color;

void main() {
	vec3 def_color = model_color;
	vec3 light = light_pos;
	vec3 p = (gl_in[0].gl_Position - gl_in[1].gl_Position).xyz;
	vec3 q = (gl_in[2].gl_Position - gl_in[1].gl_Position).xyz;
	vec3 n = normalize(cross(p, q));

	for (int i = 0; i < gl_in.length(); i++) {
	
		vec3 vertex_position_v3 = gl_in[i].gl_Position.xyz;	
		vec3 falling_light_vec = normalize(vertex_position_v3 - light);
		vec3 reverse_eye_to_vertex_vec = normalize(vertex_position_v3 - my_position);
		vec3 reflection_vec = normalize(reflect(-falling_light_vec, n));

		vec3 vision_x_refl = reverse_eye_to_vertex_vec * reflection_vec;								
		vision_x_refl = max(vision_x_refl, 0.0);

		float diffuse_coef			= length(max(falling_light_vec * n, 0.0));
		float specular_coef			= pow(length(vision_x_refl), 4);				

		vec3 vec_ambiend_color	= def_color * 0.2;
		vec3 vec_diffuse_color	= def_color * diffuse_coef  * 0.4;
		vec3 vec_specular_color = def_color * specular_coef * 1.8;

		fragColor = vec_ambiend_color + vec_diffuse_color + vec_specular_color;

		gl_Position =  mvp * model * gl_in[i].gl_Position;
		EmitVertex();
	}
	EndPrimitive();
}
