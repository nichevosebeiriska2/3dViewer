#version 430 core

layout(triangles) in;
layout(line_strip, max_vertices = 2) out;

out vec3 fragColor;
out vec3 vecNormal;

uniform vec3 light_pos;
uniform mat4 mvp;
uniform mat4 model;
uniform vec3 my_position;
uniform vec3 model_color;

vec3 CalcTriangleCenter(vec3 A, vec3 B, vec3 C)
{
	return (A+B+C)/3;
}

void main() {
	vec3 def_color = model_color;
	//vec3 light = light_pos;
	vec3 p = (gl_in[0].gl_Position - gl_in[1].gl_Position).xyz;
	vec3 q = (gl_in[2].gl_Position - gl_in[1].gl_Position).xyz;
	vec3 n = -normalize(cross(p, q));
	
	vec3 center = CalcTriangleCenter(gl_in[0].gl_Position.xyz, gl_in[1].gl_Position.xyz, gl_in[2].gl_Position.xyz);
	
	gl_Position = mvp * model * vec4(center,1.0) ;
	fragColor = model_color;
    	EmitVertex();

	gl_Position = mvp * model * vec4(center + n, 1.0);
	fragColor = model_color;
    	EmitVertex();

	EndPrimitive();
}
