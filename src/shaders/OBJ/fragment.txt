#version 430 core



uniform vec3 light_pos;
uniform mat4 mvp;
uniform mat4 model;
uniform vec3 my_position;
uniform vec3 model_color;

uniform vec3 Ka;
uniform vec3 Kd;
uniform vec3 Ks;
uniform float Ns;

uniform sampler2D u_ambientMap;
uniform sampler2D u_diffuseMap;
uniform sampler2D u_specularMap;
uniform sampler2D u_bumpMap;

uniform bool hasAmbientMap;
uniform bool hasDiffuseMap;
uniform bool hasSpecularMap;

in vec3 v_normal;
in vec3 v_fragPos;
in vec2 v_texCoord;

out vec4 outColor;

void main() 
{
;
	vec3 light = light_pos;

	vec3 falling_light_vec = normalize(v_fragPos - light);
		
	vec3 ambient = Ka;
	if(hasAmbientMap)
	{
        vec4 texColor = texture(u_ambientMap, v_texCoord);
		ambient = Ka * texColor.xyz;
	}
	
	vec3 diffuse = Kd * length(max(falling_light_vec * v_normal, 0.0));
	if(hasDiffuseMap)
	{
		vec4 texColor = texture(u_diffuseMap, v_texCoord);
		diffuse =  Kd * length(max(falling_light_vec * v_normal, 0.0)) * texColor.xyz;
	}
	
	vec3 reverse_eye_to_vertex_vec = normalize(v_fragPos - my_position);
	vec3 reflection_vec = normalize(reflect(-falling_light_vec, v_normal));

	vec3 vision_x_refl = reverse_eye_to_vertex_vec * reflection_vec;								
	vision_x_refl = max(vision_x_refl, 0.0);
		
	vec3 specular =  Ks * pow(length(vision_x_refl), 32);	
	
	if(hasSpecularMap)
	{
		vec4 texColor = texture(u_specularMap, v_texCoord);
		specular = Ks * pow(length(vision_x_refl), 32) * texColor.xyz;
	}

    vec3 result = ambient + diffuse + specular;
    outColor = vec4(result, 1.0);
}
