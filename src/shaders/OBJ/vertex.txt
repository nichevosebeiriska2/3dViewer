#version 430 core

layout(location = 0) in vec3 vertex;
layout(location = 1) in vec3 normal;
layout(location = 2) in vec2 texture_coord;

out vec3 v_normal;
out vec2 v_texCoord;
out vec3 v_fragPos;

uniform mat4 model;
uniform mat4 mvp;
uniform mat4 u_projection;

void main() {
    // Позиция вершины в мировом пространстве
    vec4 worldPos = model * vec4(vertex, 1.0);
	
    v_fragPos 	=  worldPos.xyz;
    v_normal 	= normalize(normal);
    v_texCoord 	= texture_coord;
    
    // Итоговая позиция на экране
    gl_Position = mvp * model * vec4(vertex, 1.0);
}
